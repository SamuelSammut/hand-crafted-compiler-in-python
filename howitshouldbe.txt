.main       // this is the entry point into the program
push 1      // allocate space for 1 variable (x)
oframe      // open frame - allocates space for variable
push 5      // value to store in x
push 0      // location index in the frame
push 0      // location level in the memory stack
st          // store 5 at index 0 of the frame located at level 0
push [0:0]  // push the value of x
push 5      // push the value 5 to compare
eq          // compare if x == 5, push result (1 if true, 0 if false)
push #PC+4  // push current program counter +4 (to stay in block if true)
cjmp        // conditional jump if the comparison result is true
push #PC+7  // push current program counter +7 (to skip block if false)
jmp         // unconditional jump to skip the block
push [0:0]  // push the value of x
print       // print the value of x
halt        // halt the program
